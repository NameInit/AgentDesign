    \[ \begin{pmatrix} m_{11} & m_{12} & m_{13} \\ m_{21} & m_{22} & m_{23} \\ m_{31} & m_{32} & m_{33} \end{pmatrix} \]

    \[ \begin{pmatrix} t_1 \\ t_2 \\ t_3 \end{pmatrix} \]
